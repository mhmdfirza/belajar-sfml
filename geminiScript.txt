#include <SFML/Graphics.hpp>
#include <SFML/Audio.hpp>
#include <iostream>
#include <cmath>
#include <vector>
#include <algorithm>
#include <cstdlib> // Untuk std::rand() dan std::srand()
#include <ctime>   // Untuk std::time()

using namespace std;

// Struct Peluru
struct Bullet {
    sf::CircleShape shape;
    sf::Vector2f velocity;
    bool destroy = false; // Penanda untuk dihapus
};

// Struct Musuh
struct Enemy {
    sf::CircleShape shape;
    float speed;
    bool destroy = false; // Penanda untuk dihapus
};

int main()
{
    // Seed acak berbasis waktu
    std::srand(static_cast<unsigned int>(std::time(nullptr)));

    // ----------- LOAD RESOURCES -----------
    sf::RenderWindow window(sf::VideoMode({1080, 720}), "SFML 3 - Space Shooter");
    window.setFramerateLimit(60);
    
    sf::Vector2u windowSize = window.getSize();

    // LOAD FONT
    sf::Font font;
    if (!font.openFromFile("resources/font/arial.ttf")){
        cout << "Error: Failed Load Font\n";
        return -1;
    }

    // LOAD TEXTURE
    sf::Texture texture;
    if (!texture.loadFromFile("resources/asset/cartoonshipGreen.png")){
        cout << "Error: Failed Load Asset\n";
        return -1;
    }

    // LOAD SHIP
    sf::Sprite ship(texture);
    ship.setScale({0.25f, 0.25f});
    sf::FloatRect boundsShip = ship.getLocalBounds();
    ship.setOrigin({boundsShip.size.x / 2.0f, boundsShip.size.y / 2.0f});
    ship.setPosition({
        static_cast<float>(windowSize.x) / 2.0f,
        static_cast<float>(windowSize.y) / 2.0f
    });

    // DAFTAR WADAH OBJEK
    std::vector<Bullet> bullets;
    std::vector<Enemy> enemies;

    // TIMER / CLOCK
    sf::Clock fireClock;
    const float fireRate = 0.2f; // Jeda tembakan kapal

    sf::Clock enemySpawnClock;
    const float enemySpawnRate = 1.0f; // Timbul musuh baru setiap 1 detik

    sf::Clock deltaClock;

    // LOAD TEXT
    sf::String displayString = "Mainan Tembak tembak";
    sf::Text text(font, displayString, 24);
    auto boundsText = text.getLocalBounds();
    text.setOrigin({boundsText.position.x + boundsText.size.x / 2.f, boundsText.position.y + boundsText.size.y / 2.f});
    text.setFillColor(sf::Color::White);
    text.setPosition({static_cast<float>(windowSize.x)/2.f, static_cast<float>(windowSize.y)/10.f});

    float speed = 250.f; // Kecepatan gerak kapal

    // ----------- RENDER GAME MAIN -----------
    while(window.isOpen())
    {
        float dt = deltaClock.restart().asSeconds();

        while(const std::optional event = window.pollEvent())
        {
            if(event->is<sf::Event::Closed>())
                window.close();
        }

        // 1. LOGIKA PERGERAKAN KAPAL
        if(sf::Keyboard::isKeyPressed(sf::Keyboard::Key::W))
            ship.move({0.f, -speed * dt});
        if(sf::Keyboard::isKeyPressed(sf::Keyboard::Key::S))
            ship.move({0.f, speed * dt});
        if(sf::Keyboard::isKeyPressed(sf::Keyboard::Key::A))
            ship.move({-speed * dt, 0.f});
        if(sf::Keyboard::isKeyPressed(sf::Keyboard::Key::D))
            ship.move({speed * dt, 0.f});

        // Rotasi Kapal
        if(sf::Keyboard::isKeyPressed(sf::Keyboard::Key::Left))
            ship.rotate(sf::degrees(-180.f * dt));
        if(sf::Keyboard::isKeyPressed(sf::Keyboard::Key::Right))
            ship.rotate(sf::degrees(180.f * dt));

        // 2. SPAWN PELURU (SPASI)
        if(sf::Keyboard::isKeyPressed(sf::Keyboard::Key::Space) && fireClock.getElapsedTime().asSeconds() >= fireRate)
        {
            float angleRad = ship.getRotation().asRadians();
            float bulletSpeed = 600.f;

            sf::Vector2f bulletVelocity(
                std::sin(angleRad) * bulletSpeed,
                -std::cos(angleRad) * bulletSpeed
            );

            float offset = (boundsShip.size.y * ship.getScale().y) / 2.0f;
            sf::Vector2f spawnPos = {
                ship.getPosition().x + offset * std::sin(angleRad),
                ship.getPosition().y - offset * std::cos(angleRad)
            };

            Bullet newBullet;
            newBullet.shape.setRadius(4.f);
            newBullet.shape.setFillColor(sf::Color::Yellow);
            newBullet.shape.setOrigin({4.f, 4.f});
            newBullet.shape.setPosition(spawnPos);
            newBullet.velocity = bulletVelocity;

            bullets.push_back(newBullet);
            fireClock.restart();
        }

        // 3. SPAWN ENEMY SECARA RANDOM
        if (enemySpawnClock.getElapsedTime().asSeconds() >= enemySpawnRate)
        {
            sf::Vector2f winSize = static_cast<sf::Vector2f>(window.getSize());
            sf::Vector2f spawnPos;

            // Pilih sisi spawn secara acak (0: Atas, 1: Kanan, 2: Bawah, 3: Kiri)
            int side = std::rand() % 4;
            if (side == 0)      spawnPos = { static_cast<float>(std::rand() % static_cast<int>(winSize.x)), -30.f };
            else if (side == 1) spawnPos = { winSize.x + 30.f, static_cast<float>(std::rand() % static_cast<int>(winSize.y)) };
            else if (side == 2) spawnPos = { static_cast<float>(std::rand() % static_cast<int>(winSize.x)), winSize.y + 30.f };
            else                spawnPos = { -30.f, static_cast<float>(std::rand() % static_cast<int>(winSize.y)) };

            Enemy newEnemy;
            newEnemy.shape.setRadius(15.f);
            newEnemy.shape.setFillColor(sf::Color::Red);
            newEnemy.shape.setOrigin({15.f, 15.f});
            newEnemy.shape.setPosition(spawnPos);
            newEnemy.speed = 120.f + (std::rand() % 80); // Kecepatan acak (120 - 200)

            enemies.push_back(newEnemy);
            enemySpawnClock.restart();
        }

        // 4. UPDATE GERAKAN PELURU & DETEKSI OUT OF BOUNDS
        sf::Vector2f winSize = static_cast<sf::Vector2f>(window.getSize());
        for (auto& b : bullets) {
            b.shape.move(b.velocity * dt);

            // Tandai peluru untuk dihapus jika di luar layar
            sf::Vector2f pos = b.shape.getPosition();
            if (pos.x < 0.f || pos.x > winSize.x || pos.y < 0.f || pos.y > winSize.y) {
                b.destroy = true;
            }
        }

        // 5. UPDATE GERAKAN MUSUH (MENGEJAR KAPAL)
        for (auto& e : enemies) {
            sf::Vector2f direction = ship.getPosition() - e.shape.getPosition();
            float length = std::hypot(direction.x, direction.y);
            
            if (length != 0.f) {
                direction /= length; // Normalisasi vektor arah
            }

            e.shape.move(direction * e.speed * dt);
        }

        // 6. DETEKSI BENTURAN (PELURU VS MUSUH)
        for (auto& e : enemies) {
            for (auto& b : bullets) {
                if (b.destroy || e.destroy) continue; // Skip jika sudah hancur

                // Hitung jarak antara lingkaran musuh dan lingkaran peluru
                sf::Vector2f diff = e.shape.getPosition() - b.shape.getPosition();
                float distance = std::hypot(diff.x, diff.y);
                float radiusSum = e.shape.getRadius() + b.shape.getRadius();

                // Jika jarak lebih kecil dari total jari-jari -> Bertabrakan!
                if (distance <= radiusSum) {
                    e.destroy = true;
                    b.destroy = true;
                }
            }
        }

        // 7. BERSIHKAN MEMORI (Hapus Peluru & Musuh yang "destroy == true")
        bullets.erase(
            std::remove_if(bullets.begin(), bullets.end(), [](const Bullet& b) { return b.destroy; }),
            bullets.end()
        );

        enemies.erase(
            std::remove_if(enemies.begin(), enemies.end(), [](const Enemy& e) { return e.destroy; }),
            enemies.end()
        );

        // 8. RENDER GAMBAR
        window.clear(sf::Color::Black);

        window.draw(ship);
        window.draw(text);

        // Gambar semua peluru
        for(const auto& b: bullets) {
            window.draw(b.shape);
        }

        // Gambar semua musuh
        for(const auto& e: enemies) {
            window.draw(e.shape);
        }

        window.display();
    }

    return 0;
}
